Single page app


# Basic one page app

from PyQt6.QtWidgets import (QMainWindow, QWidget, 
QVBoxLayout, QApplication)


class MainWindow(QMainWindow): # QMainWindow widget
    def __init__(self):
        super().__init__() # is necessary to inherit all atributes
        self.setWindowTitle("Your window title") # set window title

        self.main_layout = QVBoxLayout() # choose the layout

        # setting up the central widget
	# You absolutely have to have it in the one page app
	# For it to work
        self.central_widget = QWidget()
        self.central_widget.setLayout(self.main_layout)
        self.setCentralWidget(self.central_widget)
        self.setLayout(self.main_layout)


app = QApplication([])  # does not accept the command line arguments
window = MainWindow() # Creating the object
window.show() # using the inbuild show method

# executing the app
app.exec()
